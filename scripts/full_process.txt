## Merging data V4
# Feature tables
qiime feature-table merge \
  --i-tables *_asv_table.qza \
  --o-merged-table merged_asv_table.qza

qiime feature-table summarize \
  --i-table merged_asv_table.qza \
  --o-summary merged_asv_table.qzv \
  --o-sample-frequencies merged_sample_frequencies.qza \
  --o-feature-frequencies merged_asv_frequencies.qza

# Representative sequences
qiime feature-table merge-seqs \
  --i-data *_asv.qza \
  --o-merged-data merged_asv.qza

qiime feature-table tabulate-seqs \
  --i-data merged_asv.qza \
  --o-visualization merged_asv.qzv
